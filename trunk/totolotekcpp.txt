#include <iostream>
using namespace std;
int main()
{
    int n, los, tab[6];
    cout<<"Generator Lotto (6 z 49)"<<endl;
    cout<<"Podaj ilosc zakladow:"; cin>>n;
    srand(time(NULL));
    for (int x=1; x<=n; x++)
    {
    for(int i=0; i<6; i++ )
    {
    los = (rand() % 49) + 1;
    int liczbaJuzByla=0;
    for(int j=0; j<i; j++)
    if(tab[j]==los)
    liczbaJuzByla=1;
    if(liczbaJuzByla)
    i--;
    else
    tab[i] = los;
    }
    cout<<"Zaklad "<<x<<": ";
    for(int i=0; i<6; i++){
    cout<<tab[i]<<"  ";
    }
    cout<<endl;
}
    system ("pause");
    return 0;
}
